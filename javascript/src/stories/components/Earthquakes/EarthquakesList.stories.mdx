import { Meta, Story, Canvas } from "@storybook/addon-docs/blocks";
import EarthquakesList from "main/components/Earthquakes/EarthquakesList";
import earthquakeFixtures from "fixtures/earthquakeFixtures"

<Meta title="components/Earthquakes/EarthquakesList" component={EarthquakesList} />

# EarthquakesList

An `EarthquakesList` component can be used list earthquakes in a table.

export const Template = (args) => <EarthquakesList {...args} />

A table with eight earthquakes looks like this:

<Canvas>
  <Story name="EightEarthquakes" args={{ label: 'Eight Earthquakes' , data: earthquakeFixtures.eightEarthquakes }}>
    {Template.bind({})}
   </Story>
</Canvas>

A table with a single earthquake looks like this:

<Canvas>
  <Story name="OneEarthquake" args={{ label: 'One Earthquake' , data: earthquakeFixtures.oneEarthquake }}>
    {Template.bind({})}
   </Story>
</Canvas>

An empty table looks like this:

<Canvas>
  <Story name="ZeroEarthquakes" args={{ label: 'Zero Earthquakes' , data: earthquakeFixtures.zeroEarthquakes }}>
    {Template.bind({})}
   </Story>
</Canvas>

And if you pass in null for the data, you get this:

<Canvas>
  <Story name="Null" args={{ label: 'Null', data: null }}>
    {Template.bind({})}
   </Story>
</Canvas>
